    <script>
        let steps = [];
        let currentStepIndex = 0;
        
        const elInput = document.getElementById('inputExpr');
        const elSource = document.getElementById('sourceBox');
        const elStack = document.getElementById('stackBox');
        const elLog = document.getElementById('logBox');
        const elFinal = document.getElementById('finalResultBox');
        
        const btnStart = document.querySelector('.btn-start');
        const btnStep = document.getElementById('btnStep');
        const btnReset = document.getElementById('btnReset');

        function isOperator(c) {
            return ['+', '-', '*', '/', '^'].includes(c);
        }

        function log(msg) {
            elLog.innerHTML = `> ${msg}<br>` + elLog.innerHTML;
        }

        function createItem(content, type, id) {
            const div = document.createElement('div');
            div.className = `item ${type}`;
            div.textContent = content;
            if(id) div.id = id;
            return div;
        }

        // --- ALGORITHM LOGIC ---
        function generateSteps(prefix) {
            const actions = [];
            // Mô phỏng stack logic để ghi lại các bước
            let simStack = []; 

            // Duyệt từ PHẢI sang TRÁI (i giảm dần)
            for (let i = prefix.length - 1; i >= 0; i--) {
                let char = prefix[i];
                actions.push({ type: 'HIGHLIGHT', index: i, char: char });

                if (isOperator(char)) {
                    // Logic: Gặp toán tử -> Pop 2 toán hạng
                    // Quan trọng: Vì duyệt ngược, phần tử Pop đầu tiên là Op1 (Trái), phần tử thứ 2 là Op2 (Phải)
                    if (simStack.length < 2) {
                        return [{ type: 'ERROR', msg: 'Biểu thức không hợp lệ (Thiếu toán hạng)!' }];
                    }
                    
                    let op1 = simStack.pop(); // Top stack
                    let op2 = simStack.pop(); // Next top
                    
                    let newExp = `(${op1}${char}${op2})`;
                    simStack.push(newExp);

                    actions.push({ 
                        type: 'PROCESS_OP', 
                        operator: char, 
                        op1: op1, 
                        op2: op2, 
                        result: newExp,
                        msg: `Gặp toán tử '${char}'. Pop '${op1}' và '${op2}'. Tạo chuỗi: ${newExp}`
                    });

                } else {
                    // Logic: Gặp toán hạng -> Push vào Stack
                    simStack.push(char);
                    actions.push({ 
                        type: 'PUSH', 
                        value: char, 
                        msg: `Gặp toán hạng '${char}'. Push vào Stack.` 
                    });
                }
            }

            if (simStack.length === 1) {
                actions.push({ type: 'FINISH', result: simStack.pop() });
            } else {
                actions.push({ type: 'ERROR', msg: 'Biểu thức thừa toán hạng hoặc sai định dạng!' });
            }

            return actions;
        }

        // --- UI CONTROL ---

        function init() {
            let val = elInput.value.replace(/\s+/g, '');
            if(!val) return alert("Vui lòng nhập biểu thức!");

            elSource.innerHTML = '';
            elStack.innerHTML = '';
            elFinal.style.display = 'none';
            elLog.innerHTML = '';
            currentStepIndex = 0;

            // Render chuỗi nguồn lên giao diện
            for(let i = 0; i < val.length; i++) {
                elSource.appendChild(createItem(val[i], isOperator(val[i]) ? 'operator' : 'operand', `src-${i}`));
            }

            steps = generateSteps(val);
            if(steps[0] && steps[0].type === 'ERROR') {
                alert(steps[0].msg);
                return;
            }

            btnStart.style.display = 'none';
            btnStep.style.display = 'inline-block';
            btnReset.style.display = 'inline-block';
            elInput.disabled = true;
            
            log(`Bắt đầu duyệt chuỗi độ dài ${val.length} từ phải sang trái...`);
        }

        function nextStep() {
            if (currentStepIndex >= steps.length) return;

            const action = steps[currentStepIndex];

            // Xóa highlight cũ
            document.querySelectorAll('.highlight').forEach(e => e.classList.remove('highlight'));

            switch(action.type) {
                case 'HIGHLIGHT':
                    let el = document.getElementById(`src-${action.index}`);
                    if(el) {
                        el.classList.add('highlight');
                        // Scroll vào tầm nhìn nếu chuỗi dài
                        el.scrollIntoView({ behavior: 'smooth', block: 'center', inline: 'center' });
                    }
                    break;

                case 'PUSH':
                    log(action.msg);
                    let item = createItem(action.value, 'operand');
                    elStack.appendChild(item); // Append vào cuối (vì flex-direction: column-reverse hiển thị đúng tính chất stack)
                    
                    // Mark processed
                    document.querySelector('.highlight').classList.add('processed');
                    break;

                case 'PROCESS_OP':
                    log(action.msg);
                    // Visual effect: Remove 2 items from Stack UI
                    // Vì stack UI là column-reverse, phần tử mới nhất nằm ở cuối mảng children
                    let domStack = elStack.children;
                    if(domStack.length >= 2) {
                        domStack[domStack.length - 1].remove(); // Pop Op1
                        domStack[domStack.length - 1].remove(); // Pop Op2
                    }

                    // Push new complex item
                    let complexItem = createItem(action.result, 'complex');
                    elStack.appendChild(complexItem);

                    document.querySelector('.highlight').classList.add('processed');
                    break;

                case 'FINISH':
                    log("HOÀN THÀNH! Kết quả cuối cùng là duy nhất trong Stack.");
                    elFinal.textContent = `KẾT QUẢ TRUNG TỐ: ${action.result}`;
                    elFinal.style.display = 'block';
                    btnStep.disabled = true;
                    // Highlight kết quả trong stack
                    if(elStack.lastElementChild) elStack.lastElementChild.classList.add('highlight');
                    break;
                    
                case 'ERROR':
                    log(`LỖI: ${action.msg}`);
                    alert(action.msg);
                    break;
            }

            currentStepIndex++;
        }

        function reset() {
            location.reload();
        }
    </script>
